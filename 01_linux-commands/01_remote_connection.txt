# Command: psql > connect to postgresql
psql -h <ip> -p <port_num> -d <database_name>

# -h: host
# -p: port==5432 usually for postgresql service

# Command: ssh 
ssh <username>@<host>

# -L: local port forwarding
ssh -L 1234:localhost:22 user@remote.example.com
# it will listen to incoming connections on the local port 1234
ssh -L 1234:localhost:5432 christine@{target_IP}

# -D: dynamic port forwarding
ssh -D 1234 <username>@<host>

# -f -N: so we don't actuallly ssh into the box and can instead continue using that shell locally
# edit /etc/proxychains.conf
# strict_chain: not commented out
# dynaminc_chain and random_chain: commented out
socks5 127.0.0.1 1234
