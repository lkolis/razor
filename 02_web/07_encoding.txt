# Command: base64 encoding
echo "The thing I want to encode" | base64

# Command: base64 decoding
echo "The thing I want to decode" | base64 -d

# Command: hex encoding
echo "The thing I want to encode" | xxd -p

# Command: hex decoding
echo "The thing I want to decode" | xxd -p -r

# there is no Linux command for rot13

echo uggcf://jjj.unpxgurobk.rh/ | tr 'A-Za-z' 'N-ZA-Mn-za-m'
