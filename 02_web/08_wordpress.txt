----------------------------------------------------------------------
                    Plugins/themes enumeration
----------------------------------------------------------------------


# plugins enumeration
curl -s -X GET http://blog.inlanefreight.com | sed 's/href=/\n/g' | sed 's/src=/\n/g' | grep 'wp-content/plugins/*' | cut -d"'" -f2

# themes enumeration
curl -s -X GET http://blog.inlanefreight.com | sed 's/href=/\n/g' | sed 's/src=/\n/g' | grep 'themes' | cut -d"'" -f2

# plugins active enumeration
curl -I -X GET http://blog.inlanefreight.com/wp-content/plugins/someplugin

# or fuzz with a simple bash script or use a tool such as wfuzz or WPScan


----------------------------------------------------------------------
  Directory indexing: view the directory listing and convert the HTML 
----------------------------------------------------------------------

# output to a readable format with html2text
curl -s -X GET http://blog.inlanefreight.com/wp-content/plugins/mail-masta/ | html2text

# it is best practice to disable directory indexing on web servers

# shows alls posts of author with id=1 (usually its the admin user)
http://IP:PORT/?author=1

# or using curl
curl -s -I http://blog.inlanefreight.com/?author=1
